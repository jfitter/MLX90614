141 /**
142  * \brief Convert error flags to diagnostic strings and print.
143  * \param [in] err Error flags
144  */
145 void printErrStr(uint8_t err) {
146 
147  Serial.print(F(" *** "));
148  if(err == MLX90614_NORWERROR) Serial.print(F("RW Success"));
149  else {
150  Serial.print(F("Errors: "));
151  if(err & MLX90614_DATATOOLONG) Serial.print(F("Data too long / "));
152  if(err & MLX90614_TXADDRNACK) Serial.print(F("TX addr NACK / "));
153  if(err & MLX90614_TXDATANACK) Serial.print(F("TX data NACK / "));
154  if(err & MLX90614_TXOTHER) Serial.print(F("Unknown / "));
155  if(err & MLX90614_RXCRC) Serial.print(F("RX CRC / "));
156  if(err & MLX90614_INVALIDATA) Serial.print(F("Invalid data / "));
157  if(err & MLX90614_EECORRUPT) Serial.print(F("EEPROM / "));
158  if(err & MLX90614_RFLGERR) Serial.print(F("RFlags / "));
159  }
160 }
