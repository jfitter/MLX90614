72 /**
73  * \brief Return a temperature from the specified source in specified units.
74  * \remarks
75  * \li Temperature is stored in ram as a 16 bit absolute value to a resolution of 0.02K
76  * \li Linearized sensor die temperature is available as Ta (ambient).
77  * \li One or two object temperatures are linearized to the range -38.2C...125C
78  * \param [in] tsrc Internal temperature source to read, default #1.
79  * \param [in] tunit Temperature units to convert raw data to, default deg Celsius.
80  * \return Temperature.
81  */
82 double MLX90614::readTemp(tempSrc_t tsrc, tempUnit_t tunit) {
83  double temp;
84 
85  _rwError = 0;
86  switch(tsrc) {
87  case MLX90614_SRC01 : temp = read16(MLX90614_TOBJ1); break;
88  case MLX90614_SRC02 : temp = read16(MLX90614_TOBJ2); break;
89  default : temp = read16(MLX90614_TA);
90  }
91  temp *= 0.02;
92  switch(tunit) {
93  case MLX90614_TC : return convKtoC(temp);
94  case MLX90614_TF : return convKtoC(convCtoF(temp));
95  default : return temp;
96  }
97 }

127  * \brief Set the coefficients of the IIR digital filter.
128  * \remarks The IIR digital filter coefficients are set by the LS 3 bits of ConfigRegister1
129  * \n The value of the coefficients is set as follows:
130  * \n <tt> \verbatim
131  csb = 0 a1 = 0.5 a2 = 0.5
132  1 0.25 0.75
133  2 0.167 0.833
134  3 0.125 0.875
135  4 1 0 (IIR bypassed)
136  5 0.8 0.2
137  6 0.67 0.33
138  7 0.57 0.43 \endverbatim </tt>
139  * \param [in] csb See page 12 of datasheet. Range 0...7, default = 4 (IIR bypassed)
140  */
141 void MLX90614::setIIRcoeff(uint8_t csb) {
142 
143  _rwError = 0;
144 
145  // ensure legal range by clearing all but the LS 3 bits
146  csb &= 7;
147 
148  // get the current value of ConfigRegister1
149  uint16_t reg = readEEProm(MLX90614_CONFIG);
150 
151  // clear bits 2:0, mask in the new value, then write it back
152  if(!_rwError) {
153  reg &= 0xfff8;
154  reg |= (uint16_t)csb;
155  writeEEProm(MLX90614_CONFIG, reg);
156  }
157 }

263 /**
264  * \brief Return a 16 bit value read from RAM or EEPROM.
265  * \param [in] cmd Command to send (register to read from).
266  * \return Value read from memory.
267  */
268 uint16_t MLX90614::read16(uint8_t cmd) {
269  uint16_t val;
270  CRC8 crc(MLX90614_CRC8POLY);
271 
272  // send the slave address then the command and set any
273  // error status bits returned by the write
274  Wire.beginTransmission(_addr);
275  Wire.write(cmd);
276  _rwError |= (1 << Wire.endTransmission(false)) >> 1;
277 
278  // experimentally determined delay to prevent read errors
279  // (manufacturer's data sheet has left something out)
280  delayMicroseconds(MLX90614_XDLY);
281 
282  // resend slave address then get the 3 returned bytes
283  Wire.requestFrom(_addr, (uint8_t)3);
284 
285  // data is returned as 2 bytes little endian
286  val = Wire.read();
287  val |= Wire.read() << 8;
288 
289  // read the PEC (CRC-8 of all bytes)
290  _pec = Wire.read();
291 
292  // build our own CRC-8 of all received bytes
293  crc.crc8(_addr << 1);
294  crc.crc8(cmd);
295  crc.crc8((_addr << 1) + 1);
296  crc.crc8(lowByte(val));
297  _crc8 = crc.crc8(highByte(val));
298 
299  // set error status bit if CRC mismatch
300  if(_crc8 != _pec) _rwError |= MLX90614_RXCRC;
301 
302  return val;
303 }

306  * \brief Write a 16 bit value to memory.
307  * \param [in] cmd Command to send (register to write to).
308  * \param [in] data Value to write.
309  */
310 void MLX90614::write16(uint8_t cmd, uint16_t data) {
311  CRC8 crc(MLX90614_CRC8POLY);
312 
313  // build the CRC-8 of all bytes to be sent
314  crc.crc8(_addr << 1);
315  crc.crc8(cmd);
316  crc.crc8(lowByte(data));
317  _crc8 = crc.crc8(highByte(data));
318 
319  // send the slave address then the command
320  Wire.beginTransmission(_addr);
321  Wire.write(cmd);
322 
323  // write the data low byte first
324  Wire.write(lowByte(data));
325  Wire.write(highByte(data));
326 
327  // then write the crc and set the error status bits
328  Wire.write(_pec = _crc8);
329  _rwError |= (1 << Wire.endTransmission(false)) >> 1; // fixed 19jul17 - changed from "true"
330 }

339 /**
340  * \brief Write a 16 bit value to EEPROM after first clearing the memory.
341  * \remarks
342  * \li Erase and write time 5ms per manufacturer specification
343  * \li Manufacturer does not specify max or min erase/write times
344  * \param [in] reg Address to write to.
345  * \param [in] data Value to write.
346  */
347 void MLX90614::writeEEProm(uint8_t reg, uint16_t data) {
348  uint16_t val;
349  reg |= 0x20;
350 
351  // read current value, compare to the new value, and do nothing on a match
352  // or if there are read errors set the error status flag only
353  val = read16(reg);
354  if((val != data) && !_rwError) {
355 
356  // on any R/W errors it is assumed the memory is corrupted
357  // clear the memory and wait Terase (per manufacturer's documentation)
358  write16(reg, 0);
359  delay(5);
360 
361  // if no write errors then write the new value
362  if(_rwError) _rwError |= MLX90614_EECORRUPT;
363  else {
364 
365  // write the data and wait Twrite (per manufacturer's documentation)
366  write16(reg, data);
367  delay(5);
368  if(_rwError) _rwError |= MLX90614_EECORRUPT;
369  }
370  }
371 }
