74 /**
75  * \brief Return a temperature from the specified source in specified units.
76  * \remarks
77  * \li Temperature is stored in ram as a 16 bit absolute value to a resolution of 0.02K
78  * \li Linearized sensor die temperature is available as Ta (ambient).
79  * \li One or two object temperatures are linearized to the range -38.2C...125C
80  * \param [in] tsrc Internal temperature source to read, default #1.
81  * \param [in] tunit Temperature units to convert raw data to, default deg Celsius.
82  * \return Temperature.
83  */
84 double MLX90614::readTemp(tempSrc_t tsrc, tempUnit_t tunit) {
85  double temp;
86 
87  _rwError = 0;
88  switch(tsrc) {
89  case MLX90614_SRC01 : temp = read16(MLX90614_TOBJ1); break;
90  case MLX90614_SRC02 : temp = read16(MLX90614_TOBJ2); break;
91  default : temp = read16(MLX90614_TA);
92  }
93  temp *= 0.02;
94  switch(tunit) {
95  case MLX90614_TC : return convKtoC(temp);
96  case MLX90614_TF : return convKtoC(convCtoF(temp));
97  }
98  return temp;
99 }

129  * \brief Set the coefficients of the IIR digital filter.
130  * \remarks The IIR digital filter coefficients are set by the LS 3 bits of ConfigRegister1
131  * \n The value of the coefficients is set as follows:
132  * \n <tt> \verbatim
133  csb = 0 a1 = 0.5 a2 = 0.5
134  1 0.25 0.75
135  2 0.167 0.833
136  3 0.125 0.875
137  4 1 0 (IIR bypassed)
138  5 0.8 0.2
139  6 0.67 0.33
140  7 0.57 0.43 \endverbatim </tt>
141  * \param [in] csb See page 12 of datasheet. Range 0...7, default = 4 (IIR bypassed)
142  */
143 void MLX90614::setIIRcoeff(uint8_t csb) {
144 
145  _rwError = 0;
146 
147  // Ensure legal range by clearing all but the LS 3 bits.
148  csb &= 7;
149 
150  // Get the current value of ConfigRegister1
151  uint16_t reg = readEEProm(MLX90614_CONFIG);
152 
153  // Clear bits 2:0, mask in the new value, then write it back.
154  if(!_rwError) {
155  reg &= 0xfff8;
156  reg |= (uint16_t)csb;
157  writeEEProm(MLX90614_CONFIG, reg);
158  }
159 }

221  * \brief Set device SMBus address.
222  * \remarks
223  * \li Must be only device on the bus.
224  * \li Must power cycle the device after changing address.
225  * \param [in] a New device address. Range 1...127
226  */
227 void MLX90614::setAddr(uint8_t addr) {
228 
229  _rwError = 0;
230 
231  // It is assumed we do not know the existing slave address so the broadcast address is used.
232  // First ensure the new address is in the legal range (1..127)
233  if(addr &= 0x7f) {
234  _addr = MLX90614_BROADCASTADDR;
235  writeEEProm(MLX90614_ADDR, addr);
236 
237  // There will always be a r/w error using the broadcast address so we cannot respond
238  // to r/w errors. We must just assume this worked.
239  _addr = addr;
240 
241  } else _rwError |= MLX90614_INVALIDATA;
242 }

266 /**
267  * \brief Return a 16 bit value read from RAM or EEPROM.
268  * \param [in] cmd Command to send (register to read from).
269  * \return Value read from memory.
270  */
271 uint16_t MLX90614::read16(uint8_t cmd) {
272  uint16_t val;
273  CRC8 crc(MLX90614_CRC8POLY);
274 
275  // Send the slave address then the command and set any error status bits returned by the write.
276  Wire.beginTransmission(_addr);
277  Wire.write(cmd);
278  _rwError |= (1 << Wire.endTransmission(false)) >> 1;
279 
280  // Experimentally determined delay to prevent read errors (manufacturer's data sheet has
281  // left something out).
282  delayMicroseconds(MLX90614_XDLY);
283 
284  // Resend slave address then get the 3 returned bytes.
285  Wire.requestFrom(_addr, (uint8_t)3);
286 
287  // Data is returned as 2 bytes little endian.
288  val = Wire.read();
289  val |= Wire.read() << 8;
290 
291  // Rread the PEC (CRC-8 of all bytes).
292  _pec = Wire.read();
293 
294  // Clear r/w errors if using broadcast address.
295  if(_addr == MLX90614_BROADCASTADDR) _rwError &= MLX90614_NORWERROR;
296 
297  // Build our own CRC-8 of all received bytes.
298  crc.crc8(_addr << 1);
299  crc.crc8(cmd);
300  crc.crc8((_addr << 1) + 1);
301  crc.crc8(lowByte(val));
302  _crc8 = crc.crc8(highByte(val));
303 
304  // Set error status bit if CRC mismatch.
305  if(_crc8 != _pec) _rwError |= MLX90614_RXCRC;
306 
307  return val;
308 }

310 /**
311  * \brief Write a 16 bit value to memory.
312  * \param [in] cmd Command to send (register to write to).
313  * \param [in] data Value to write.
314  */
315 void MLX90614::write16(uint8_t cmd, uint16_t data) {
316  CRC8 crc(MLX90614_CRC8POLY);
317 
318  // Build the CRC-8 of all bytes to be sent.
319  crc.crc8(_addr << 1);
320  crc.crc8(cmd);
321  crc.crc8(lowByte(data));
322  _crc8 = crc.crc8(highByte(data));
323 
324  // Send the slave address then the command.
325  Wire.beginTransmission(_addr);
326  Wire.write(cmd);
327 
328  // Write the data low byte first.
329  Wire.write(lowByte(data));
330  Wire.write(highByte(data));
331 
332  // Then write the crc and set the r/w error status bits.
333  Wire.write(_pec = _crc8);
334  _rwError |= (1 << Wire.endTransmission(true)) >> 1;
335 
336  // Clear r/w errors if using broadcast address.
337  if(_addr == MLX90614_BROADCASTADDR) _rwError &= MLX90614_NORWERROR;
338 }

347 /**
348  * \brief Write a 16 bit value to EEPROM after first clearing the memory.
349  * \remarks
350  * \li Erase and write time 5ms per manufacturer specification
351  * \li Manufacturer does not specify max or min erase/write times
352  * \param [in] reg Address to write to.
353  * \param [in] data Value to write.
354  */
355 void MLX90614::writeEEProm(uint8_t reg, uint16_t data) {
356  uint16_t val;
357  reg |= 0x20;
358 
359  // Read current value, compare to the new value, and do nothing on a match or if there are
360  // read errors set the error status flag only.
361  val = read16(reg);
362  if((val != data) && !_rwError) {
363 
364  // On any R/W errors it is assumed the memory is corrupted.
365  // Clear the memory and wait Terase (per manufacturer's documentation).
366  write16(reg, 0);
367  delay(5);
368  if(_rwError) _rwError |= MLX90614_EECORRUPT;
369 
370  // Write the data and wait Twrite (per manufacturer's documentation)
371  // and set the r/w error status bits.
372  write16(reg, data);
373  delay(5);
374  if(_rwError) _rwError |= MLX90614_EECORRUPT;
375  }
376 }
