74 /**
75  * \brief Return a temperature from the specified source in specified units.
76  * \remarks
77  * \li Temperature is stored in ram as a 16 bit absolute value to a resolution of 0.02K
78  * \li Linearized sensor die temperature is available as Ta (ambient).
79  * \li One or two object temperatures are linearized to the range -38.2C...125C
80  * \param [in] tsrc Internal temperature source to read, default #1.
81  * \param [in] tunit Temperature units to convert raw data to, default deg Celsius.
82  * \return Temperature.
83  */
84 double MLX90614::readTemp(tempSrc_t tsrc, tempUnit_t tunit) {
85  double temp;
86 
87  _rwError = 0;
88  switch(tsrc) {
89  case MLX90614_SRC01 : temp = read16(MLX90614_TOBJ1); break;
90  case MLX90614_SRC02 : temp = read16(MLX90614_TOBJ2); break;
91  default : temp = read16(MLX90614_TA);
92  }
93  temp *= 0.02;
94  switch(tunit) {
95  case MLX90614_TC : return convKtoC(temp);
96  case MLX90614_TF : return convKtoC(convCtoF(temp));
97  }
98  return temp;
99 }

128 /**
129  * \brief Set the coefficients of the IIR digital filter.
130  * \remarks The IIR digital filter coefficients are set by the LS 3 bits of ConfigRegister1
131  * \n The value of the coefficients is set as follows:
132  * \n <tt> \verbatim
133  csb = 0 a1 = 0.5 a2 = 0.5
134  1 0.25 0.75
135  2 0.167 0.833
136  3 0.125 0.875
137  4 1 0 (IIR bypassed)
138  5 0.8 0.2
139  6 0.67 0.33
140  7 0.57 0.43 \endverbatim </tt>
141  * \param [in] csb See page 12 of datasheet. Range 0...7, default = 4 (IIR bypassed)
142  */
143 void MLX90614::setIIRcoeff(uint8_t csb) {
144 
145  _rwError = 0;
146 
147  // Ensure legal range by clearing all but the LS 3 bits.
148  csb &= 7;
149 
150  // Get the current value of ConfigRegister1
151  uint16_t reg = readEEProm(MLX90614_CONFIG);
152 
153  // Clear bits 2:0, mask in the new value, then write it back.
154  if(!_rwError) {
155  reg &= 0xfff8;
156  reg |= (uint16_t)csb;
157  writeEEProm(MLX90614_CONFIG, reg);
158  }
159 }

219 /**
220  * \brief Set device SMBus address.
221  * \remarks
222  * \li Must be only device on the bus.
223  * \li Must power cycle the device after changing address.
224  * \param [in] addr New device address. Range 1...127
225  */
226 void MLX90614::setAddr(uint8_t addr) {
227 
228  _rwError = 0;
229 
230  // It is assumed we do not know the existing slave address so the broadcast address is used.
231  // First ensure the new address is in the legal range (1..127)
232  if(addr &= 0x7f) {
233  _addr = MLX90614_BROADCASTADDR;
234  writeEEProm(MLX90614_ADDR, addr);
235 
236  // There will always be a r/w error using the broadcast address so we cannot respond
237  // to r/w errors. We must just assume this worked.
238  _addr = addr;
239 
240  } else _rwError |= MLX90614_INVALIDATA;
241 }

246  * \li Must be only device on the bus.
247  * \li Sets the library to use the new found address.
248  * \return Device address.

252 
253  _rwError = 0;
254

265 /**
266  * \brief Return a 16 bit value read from RAM or EEPROM.
267  * \param [in] cmd Command to send (register to read from).
268  * \return Value read from memory.
269  */
270 uint16_t MLX90614::read16(uint8_t cmd) {
271  uint16_t val;
272  CRC8 crc(MLX90614_CRC8POLY);
273 
274  // Send the slave address then the command and set any error status bits returned by the write.
275  Wire.beginTransmission(_addr);
276  Wire.write(cmd);
277  _rwError |= (1 << Wire.endTransmission(false)) >> 1;
278 
279  // Experimentally determined delay to prevent read errors (manufacturer's data sheet has
280  // left something out).
281  delayMicroseconds(MLX90614_XDLY);
282 
283  // Resend slave address then get the 3 returned bytes.
284  Wire.requestFrom(_addr, (uint8_t)3);
285 
286  // Data is returned as 2 bytes little endian.
287  val = Wire.read();
288  val |= Wire.read() << 8;
289 
290  // Rread the PEC (CRC-8 of all bytes).
291  _pec = Wire.read();
292 
293  // Clear r/w errors if using broadcast address.
294  if(_addr == MLX90614_BROADCASTADDR) _rwError &= MLX90614_NORWERROR;
295 
296  // Build our own CRC-8 of all received bytes.
297  crc.crc8(_addr << 1);
298  crc.crc8(cmd);
299  crc.crc8((_addr << 1) + 1);
300  crc.crc8(lowByte(val));
301  _crc8 = crc.crc8(highByte(val));
302 
303  // Set error status bit if CRC mismatch.
304  if(_crc8 != _pec) _rwError |= MLX90614_RXCRC;
305 
306  return val;
307 }

309 /**
310  * \brief Write a 16 bit value to memory.
311  * \param [in] cmd Command to send (register to write to).
312  * \param [in] data Value to write.
313  */
314 void MLX90614::write16(uint8_t cmd, uint16_t data) {
315  CRC8 crc(MLX90614_CRC8POLY);
316 
317  // Build the CRC-8 of all bytes to be sent.
318  crc.crc8(_addr << 1);
319  crc.crc8(cmd);
320  crc.crc8(lowByte(data));
321  _crc8 = crc.crc8(highByte(data));
322 
323  // Send the slave address then the command.
324  Wire.beginTransmission(_addr);
325  Wire.write(cmd);
326 
327  // Write the data low byte first.
328  Wire.write(lowByte(data));
329  Wire.write(highByte(data));
330 
331  // Then write the crc and set the r/w error status bits.
332  Wire.write(_pec = _crc8);
333  _rwError |= (1 << Wire.endTransmission(true)) >> 1;
334 
335  // Clear r/w errors if using broadcast address.
336  if(_addr == MLX90614_BROADCASTADDR) _rwError &= MLX90614_NORWERROR;
337 }

346 /**
347  * \brief Write a 16 bit value to EEPROM after first clearing the memory.
348  * \remarks
349  * \li Erase and write time 5ms per manufacturer specification
350  * \li Manufacturer does not specify max or min erase/write times
351  * \param [in] reg Address to write to.
352  * \param [in] data Value to write.
353  */
354 void MLX90614::writeEEProm(uint8_t reg, uint16_t data) {
355  uint16_t val;
356  reg |= 0x20;
357 
358  // Read current value, compare to the new value, and do nothing on a match or if there are
359  // read errors set the error status flag only.
360  val = read16(reg);
361  if((val != data) && !_rwError) {
362 
363  // On any R/W errors it is assumed the memory is corrupted.
364  // Clear the memory and wait Terase (per manufacturer's documentation).
365  write16(reg, 0);
366  delay(5);
367  if(_rwError) _rwError |= MLX90614_EECORRUPT;
368 
369  // Write the data and wait Twrite (per manufacturer's documentation)
370  // and set the r/w error status bits.
371  write16(reg, data);
372  delay(5);
373  if(_rwError) _rwError |= MLX90614_EECORRUPT;
374  }
375 }
